const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // UI Elements const scoreDisplay = document.getElementById('score'); const levelDisplay = document.getElementById('level'); const livesDisplay = document.getElementById('lives'); const currentWordDisplay = document.getElementById('currentWordDisplay'); const targetWordDisplay = document.getElementById('targetWordDisplay'); const messageDisplay = document.getElementById('messageDisplay'); const startButton = document.getElementById('startButton'); const restartButton = document.getElementById('restartButton'); const highScoresList = document.getElementById('highScoresList'); // Game Settings let player, bullets, fallingLetters, score, level, lives, currentWord, targetWords, letterSpeed, letterSpawnRate; let gameInterval; let gameActive = false; const playerWidth = 50; const playerHeight = 20; const playerSpeed = 15; const bulletSpeed = 7; const bulletRadius = 5; const letterSize = 20; let maxLives = 3; // Player movement const keysPressed = {}; const levelSettings = [ { words: ["CAT", "DOG", "SUN"], speed: 0.5, spawn: 1500 }, // Level 1 { words: ["CODE", "GAME", "HTML", "FUN"], speed: 0.75, spawn: 1200 }, // Level 2 { words: ["PYTHON", "GITHUB", "LEVELS", "AWESOME"], speed: 1, spawn: 1000 }, // Level 3 { words: ["JAVASCRIPT", "DEVELOPER", "ALGORITHM", "PROGRAMMING"], speed: 1.25, spawn: 900 }, // Level 4 { words: ["FRAMEWORK", "DATABASE", "APPLICATION", "INTEGRATION"], speed: 1.5, spawn: 800 } // Level 5 ]; let currentLevelWords = []; let currentTargetWord = ""; const highScores = JSON.parse(localStorage.getItem('letterFallHighScores')) || []; function updateHighScoresDisplay() { highScoresList.innerHTML = ''; if (highScores.length === 0) { highScoresList.innerHTML = '
  • No scores yet!
    • '; return; } highScores.sort((a, b) => b.score - a.score); // Sort descending const topScores = highScores.slice(0, 5); // Show top 5 topScores.forEach(scoreItem => { const li = document.createElement('li'); li.textContent = `${scoreItem.name} - ${scoreItem.score}`; highScoresList.appendChild(li); }); } function addHighScore(name, score) { highScores.push({ name, score }); localStorage.setItem('letterFallHighScores', JSON.stringify(highScores)); updateHighScoresDisplay(); } function initGame() { player = { x: canvas.width / 2 - playerWidth / 2, y: canvas.height - playerHeight - 10, width: playerWidth, height: playerHeight, color: '#007bff' // Blue }; bullets = []; fallingLetters = []; score = 0; level = 1; lives = maxLives; currentWord = ""; setupLevel(); updateUI(); if (gameInterval) clearInterval(gameInterval); gameInterval = setInterval(gameLoop, 1000 / 60); // 60 frames per second gameActive = true; messageDisplay.style.display = 'none'; startButton.style.display = 'none'; restartButton.style.display = 'none'; canvas.focus(); // Ensure canvas can receive keyboard input } function setupLevel() { const settings = levelSettings[level - 1]; if (!settings) { gameOver(true); // Won the game return; } currentLevelWords = [...settings.words]; letterSpeed = settings.speed; letterSpawnRate = settings.spawn; pickNewTargetWord(); levelDisplay.textContent = level; } function pickNewTargetWord() { if (currentLevelWords.length === 0) { levelUp(); return; } const randomIndex = Math.floor(Math.random() * currentLevelWords.length); currentTargetWord = currentLevelWords.splice(randomIndex, 1)[0]; targetWordDisplay.textContent = currentTargetWord; currentWord = ""; // Reset current word when a new target is set currentWordDisplay.textContent = ""; } function levelUp() { level++; showMessage(`Level ${level}!`, 1500, '#00C853'); // Green for level up if (level > levelSettings.length) { gameOver(true); // Player won all levels } else { setupLevel(); score += 100 * (level - 1); // Bonus for leveling up updateUI(); } } function updateUI() { scoreDisplay.textContent = score; levelDisplay.textContent = level; livesDisplay.textContent = lives; currentWordDisplay.textContent = currentWord; } function drawPlayer() { ctx.fillStyle = player.color; ctx.fillRect(player.x, player.y, player.width, player.height); // Add a little "cannon" ctx.fillStyle = '#333'; ctx.fillRect(player.x + player.width / 2 - 5, player.y - 10, 10, 10); } function drawBullets() { bullets.forEach(bullet => { ctx.beginPath(); ctx.arc(bullet.x, bullet.y, bullet.radius, 0, Math.PI * 2); ctx.fillStyle = bullet.color; ctx.fill(); ctx.closePath(); }); } function drawFallingLetters() { fallingLetters.forEach(letterObj => { ctx.fillStyle = letterObj.color; ctx.font = `${letterSize}px 'Courier New', Courier, monospace`; ctx.fillText(letterObj.char, letterObj.x, letterObj.y); }); } function spawnLetter() { if (!gameActive || !currentTargetWord) return; let charToSpawn; // Increase chance of spawning a needed letter if (currentWord.length < currentTargetWord.length && Math.random() < 0.6) { // 60% chance for a needed letter // Try to spawn the next letter in the target word charToSpawn = currentTargetWord[currentWord.length]; // If that letter is already in currentWord (meaning it's not the first occurrence needed) // or if it's the last letter and it's already in the currentWord and we've already gotten to this spot, // then fall back to a random needed letter. if (currentWord.includes(charToSpawn) && currentWord.indexOf(charToSpawn) <= currentTargetWord.indexOf(charToSpawn)) { const neededLetters = currentTargetWord.split('').filter(c => !currentWord.includes(c) || currentWord.split(c).length <= currentTargetWord.split(c).length); if (neededLetters.length > 0) { charToSpawn = neededLetters[Math.floor(Math.random() * neededLetters.length)]; } else { // All letters are technically "in" currentWord, but maybe not in correct count/order // Fallback to a random letter from the target word if all needed are 'gotten' charToSpawn = currentTargetWord[Math.floor(Math.random() * currentTargetWord.length)]; } } } else { // 40% chance for a random letter (distractor or useful) const alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"; charToSpawn = alphabet[Math.floor(Math.random() * alphabet.length)]; } const x = Math.random() * (canvas.width - letterSize); const y = 0; const colors = ['#ff0000', '#00ff00', '#0000ff', '#ff00ff', '#ffa500', '#00ffff']; // Red, Green, Blue, Magenta, Orange, Cyan const color = colors[Math.floor(Math.random() * colors.length)]; fallingLetters.push({ char: charToSpawn, x, y, color }); } let lastSpawnTime = 0; function gameLoop() { if (!gameActive) return; ctx.clearRect(0, 0, canvas.width, canvas.height); // Handle player movement if (keysPressed['ArrowLeft'] || keysPressed['a']) { player.x = Math.max(0, player.x - playerSpeed); } if (keysPressed['ArrowRight'] || keysPressed['d']) { player.x = Math.min(canvas.width - player.width, player.x + playerSpeed); } drawPlayer(); drawBullets(); drawFallingLetters(); // Move Bullets bullets.forEach((bullet, bIndex) => { bullet.y -= bulletSpeed; if (bullet.y < 0) { bullets.splice(bIndex, 1); } }); // Move Letters & Collision Detection for (let lIndex = fallingLetters.length - 1; lIndex >= 0; lIndex--) { const letter = fallingLetters[lIndex]; letter.y += letterSpeed; // Letter reaches bottom if (letter.y > canvas.height) { fallingLetters.splice(lIndex, 1); lives--; updateUI(); if (lives <= 0) { gameOver(false); } continue; // Continue to next letter as current one was removed } // Bullet-Letter Collision for (let bIndex = bullets.length - 1; bIndex >= 0; bIndex--) { const bullet = bullets[bIndex]; const dist = Math.sqrt( (bullet.x - (letter.x + letterSize / 2)) ** 2 + (bullet.y - (letter.y - letterSize / 2)) ** 2 ); if (dist < bullet.radius + letterSize / 2) { // Hit! bullets.splice(bIndex, 1); // Remove bullet fallingLetters.splice(lIndex, 1); // Remove letter // If the hit letter is the next correct letter in the target word if (currentTargetWord.startsWith(currentWord + letter.char)) { currentWord += letter.char; currentWordDisplay.textContent = currentWord; score += 10; // Points for hitting a correct letter // Check if currentWord matches targetWord if (currentWord === currentTargetWord) { score += 50; // Bonus for completing word showMessage(`"${currentTargetWord}" formed! +50`, 1500, '#4CAF50'); pickNewTargetWord(); } } else { // Incorrect letter or out of sequence showMessage(`Oops! Wrong letter for "${currentTargetWord}"`, 1000, '#f44336'); currentWord = ""; // Reset current word if sequence is broken currentWordDisplay.textContent = ""; lives--; // Penalty for wrong letter if (lives <= 0) { gameOver(false); } } updateUI(); break; // Break from bullet loop as letter is removed } } } // Spawn new letters const currentTime = Date.now(); if (currentTime - lastSpawnTime > letterSpawnRate) { spawnLetter(); lastSpawnTime = currentTime; } } function shoot() { if (!gameActive) return; const bullet = { x: player.x + player.width / 2, y: player.y - playerHeight / 2, radius: bulletRadius, color: '#ffc107' // Yellow bullet }; bullets.push(bullet); } function showMessage(msg, duration = 2000, color = 'white') { messageDisplay.textContent = msg; messageDisplay.style.backgroundColor = color; messageDisplay.style.display = 'block'; setTimeout(() => { messageDisplay.style.display = 'none'; }, duration); } function gameOver(win) { gameActive = false; clearInterval(gameInterval); let finalMessage = ""; let messageColor = ''; if (win) { finalMessage = "ðŸŽ‰ YOU WON! ðŸŽ‰\nAll levels completed!"; messageColor = '#00C853'; // Green } else { finalMessage = "GAME OVER!\nTry again?"; messageColor = 'rgba(244, 67, 54, 0.9)'; // Red } messageDisplay.textContent = finalMessage; messageDisplay.style.backgroundColor = messageColor; messageDisplay.style.display = 'block'; // Prompt for name and add high score setTimeout(() => { const playerName = prompt("Enter your name for the high score list:"); if (playerName) { addHighScore(playerName, score); } restartButton.style.display = 'block'; }, win ? 2000 : 1000); // Shorter delay for game over, longer for win } // Event Listeners startButton.addEventListener('click', initGame); restartButton.addEventListener('click', () => { initGame(); restartButton.style.display = 'none'; startButton.style.display = 'none'; // Ensure start button stays hidden }); document.addEventListener('keydown', (e) => { keysPressed[e.key] = true; if (e.key === ' ' && gameActive) { // Spacebar to shoot shoot(); } }); document.addEventListener('keyup', (e) => { keysPressed[e.key] = false; }); // Initial setup updateHighScoresDisplay(); targetWordDisplay.textContent = "START"; // Initial display
